\begin{SongText}[Byssan lull]
    \begin{SongInfo}
        Mel: Byssan lull
    \end{SongInfo}
    \begin{SongVerse}
        $\:\|$ Byssan lull utav brännvin blir man full,\\*%
        och slipsen man doppar i smöret. $\:\|$\\*%
        Ja, näsan den blir röd,\\*%
        och ögonen får glöd,\\*%
        och tusan så bra blir humöret.
    \end{SongVerse}
\end{SongText}
