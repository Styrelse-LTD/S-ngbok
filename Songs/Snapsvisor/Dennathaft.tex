\begin{SongText}[Denna thaft]
    \begin{SongInfo}
        Mel: Helan
    \end{SongInfo}
    \begin{SongVerse}
        Denna thaft\\*%
        är den bätha thaft thythemet haft.\\*%
        Denna thaft\\*%
        är den bätha thaft dom haft.\\*%
        Och den thom inte har nån kraft,\\*%
        han dricka thkall av denna thaft.\\*%
        Denna thaft, till landth, till sjöth, till havth.
    \end{SongVerse}
\end{SongText}
